CREATE TABLE  E_BOOK
   (	CALLNO VARCHAR(40), 
	NAME VARCHAR(255), 
	AUTHOR VARCHAR(255), 
	PUBLISHER VARCHAR(255), 
	QUANTITY INT, 
	ISSUED INT, 
	 CONSTRAINT E_BOOK_PK PRIMARY KEY (CALLNO)
   );


CREATE TABLE  E_LIBRARIAN
   (	ID INT AUTO_INCREMENT, 
	NAME VARCHAR(255), 
	PASSWORD VARCHAR(255), 
	EMAIL VARCHAR(255), 
	MOBILE INT, 
	 CONSTRAINT E_LIBRARIAN_PK PRIMARY KEY (ID)
   );


Note: ID must be generated through sequence in E_LIBRARIAN table.

CREATE TABLE  E_ISSUEBOOK
   (	CALLNO VARCHAR(40) NOT NULL, 
	STUDENTID VARCHAR(255) NOT NULL, 
	STUDENTNAME VARCHAR(255), 
	STUDENTMOBILE INT, 
	ISSUEDDATE DATE, 
	RETURNSTATUS VARCHAR(250)
   );
